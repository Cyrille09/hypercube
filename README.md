```bash
### Clone the repo

git clone https://github.com/Cyrille09/hypercube.git

########### PYTHON INSTALLATION ###########

### redirect to the python task folder with the below command
cd flash-server

# run the task
python3 server.py

########## REACT.JS INSTALLATION ##########

### redirect to the react.js task folder with the below command
cd client

### Install the dependencies
npm install

# Create .env file and add the following info
PORT=3007
REACT_APP_API_URL="http://127.0.0.1:5000"

# run the task
npm start

```
